1.5. compare the sympy vs maxima results

def p1(E, SN):
  """
  sympy result
  """
  return (-1 - (-E - 12*SN)/(12*SN))**(1/3) + 1


def p2(E, SN):
  """
  maxima result
  """
  SDR = (4**((-1.0E+0)/3.0E+0)*9**(1.0E+0/3.0E+0)*E**(1.0E+0/3.0E+0)*SN**((-1.0E+0)/3.0E+0)+3)/3.0E+0
  return SDR


for i in range(1000):
  for j in range(1000):
    i = i+1
    j = j+1
    diff = p1(i,j)-p2(i,j)
    if diff > 0.00001:
      print('error')
